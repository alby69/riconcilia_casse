# 1. Inizializza un repository Git locale
git init -b main

# 2. Aggiungi tutti i file al monitoraggio (esclusi quelli nel .gitignore)
git add .

# 3. Crea la prima "fotografia" del tuo progetto (commit)
git commit -m "Initial commit: Aggiunto sistema di riconciliazione contabile"

# 4. Collega il tuo repository locale a quello remoto su GitHub
#    Sostituisci <TUO_USERNAME> con il tuo nome utente GitHub
git remote add origin https://github.com/<TUO_USERNAME>/riconcilia_casse.git

# 5. Carica i file su GitHub
git push -u origin main
